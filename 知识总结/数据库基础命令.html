数据库培训

一、 基础知识

  1. 基础常用命令
    • mysql -uroot -p密码;(也可以不带密码，之后输入) 本地登录
    • mysql -h 登录ip -p 端口(通常3306） -uroot -p密码; 远程登录
    • show databases; 查数据库
    • use database; 切换数据库
    • show tables; 看表
    • desc table; 查看表结构信息
    • create database 数据库名 default charset utf8; 创建数据库
    • create table 表名(列名 数据类型 约束···，列名 数据类型 约束···）engine=innodb default charset=utf8 创建表
    • 其中数据类型种类 数字(int，tinyint,smallint,float,double),字符串(char(个数)varchar(个数))时间(DATE,TIME,DATETIME),枚举enum(值只能是枚举中的元素），集合set(值只能是结合元素的组合)
    • create table 表名(列名 数据类型 primary key auto_increment,列名 数据类型 约束···）engine=innodb default charset=utf8 创建含主键的表
  2. 查询命令
    • 条件查询
      • select * from 表名;
      • select 列名···from 表名;
      • select 列名 from 表名 where 列名(id等) >/</!= value;
      • select 列名，常量 from 表名; 增加一个常量列
      • select 列名 from 表名 where 列名 in/not in/between and value;
      • select 列名 from 表名 where 条件1 and 条件2;
      • 特殊的：select 列名 from 表名 where 列名 in (select 列名(只能一列) from 表名);
      • select 列名 from 表名 where 列名 like 'xx%'/'%xx'/"xx_"; 查询以xx开头/xx结尾 %代表任意位，_代表一位
      • select xx,(select xx from ...) from ..... 查询条件做常量值
    • 分页
      • select 列名 from 表名 limit num; 显示num个
      • select 列名 from 表名 limit num1,num2;从num1后取num2行数据，num1是起始位置，num2是个数
      • select 列名 from 表名 limit num1 offset num2;从num2后取num1行数据，num2是起始位置，num1是个数
    • 排序
      • select * from 表名 order by 列名 desc;从大到小排序
      • select * from 表名 order by 列名 asc;从小到大排序
      • select * from 表名 order by 列名1 desc 列名2 asc; 首先遵循列1从大到小排序，遇到相同数据时，按列2从小到大排序
    • 分组操作
      • select count/sum/max/min/avg(列名1），列名2 from 表名 group by 列名(通常是列名2）；分组操作
      • select count/sum/max/min/avg(列名1），列名2 from 表名 group by 列名(通常是列名2）having 条件；分组操作后筛选
    • 连表操作
      • select * from 表1 left join 表2 on 表1.列名=表2.列名; 左连接
      • select * from 表1 right join 表2 on 表1.列名=表2.列名; 右连接
      • select * from 表1 inner join 表2 on 表1.列名=表2.列名; 内连接
    • 临时表
      • (select * from 表名)as e
  3. 删除，修改，插入命令
    • 插入
      • insert into 表名(列名1，列名2···) values(值1，值2···),(值1，值2···),(值1，值2···); 插入值
      • insert into 表名1(列名) select 列名 from 表2; 在一个表中插入另一个表中数据
    • 修改
      • update 表名 set 列名1=value1，列名2=value2 where 条件1 [and/or 条件2]
    • 删除
      • delete from 表名; 清除表(如果有自增id，id 不会重新开始)
      • delete from 表名 where 条件； 清除特定数据
      • truncate table 表名;清除表(如果有自增id，id 会重新开始)
  4. 修改表结构
    • alter table 表名 auto_increment=value;设置自增键起始值；
    • alter table 表名 drop 列名;删除列
    • alter table 表名 add 列名 数据类型 约束; 增加列
    • alter table 表名 change 旧列名 新列名 数据类型; 修改字段类型
    • alter table 表名 modify 列名 数据类型; 修改数据类型
    • alter table 旧表名 rename 新表名; 修改表名
    • alter table 表名 drop primary key; 删除表中主键
    • alter table 表名 add 列名 数据类型 primary key;添加主键
    • alter table 表名 add primary key(列名);设置主键
    • alter table 表名 add column 列名 数据类型 after 列名;在某一列后添加主键
  5. 索引
    • 索引种类
      • 普通索引:加速查找
      • 主键索引:加速查找；约束(主键不能为空，不能重复)
      • 唯一索引:加速查找；约束(索引不能重复）
      • 组合索引:多列组成一个索引(联合主键索引，联合唯一索引，联合普通索引)
    • 常用命令
      • create index 索引名 on 表名(列名[,列名]);创建[联合]普通索引
      • create unique index 索引名 on 表名(列名);创建唯一索引
      • create unique index 索引名 on 表名(列名,列名);创建联合唯一索引
      • drop index 索引名 on 表名; 删除索引
  6. 权限表示例（RBAC)

    • 用户表

      • 创建表
      DROP TABLE IF EXISTS `t_user`;
CREATE TABLE `t_user` (
  `user_id` BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '用户标识',
  `user_name` varchar(50) NOT NULL COMMENT '用户名称',
  `user_password` varchar(128) NOT NULL COMMENT '用户密码',
  `user_salt` varchar(32) NULL COMMENT '密码盐',
  `create_time` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  `modify_time` timestamp NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `status` int(2) NOT NULL DEFAULT '1' COMMENT '状态',
  `last_login_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
      • 新增数据
      insert into t_user(user_name, user_password) values('admin','123456');
insert into t_user(user_name, user_password, last_login_time) values('admin','123456',NOW());
      • 删除数据
      delete from t_user where user_id = 1;
      • 修改数据
      update t_user set user_password = '234567' where user_id = 1;
update t_user set status = 1 where user_id = 1;
      • 查询数据
      select * from t_user where user_id = 1 and status = 1;

    • 角色表

      • 创建表
      DROP TABLE IF EXISTS `t_role`;
CREATE TABLE `t_role` (
`role_id` BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '角色标识',
`role_code` VARCHAR(20) NOT NULL COMMENT '角色编码',
`role_name` VARCHAR(50) NOT NULL COMMENT '角色名称',
`remark` VARCHAR(150) COMMENT '备注',
`create_time` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
`modify_time` timestamp NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间'
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT '角色表';
      • 新增唯一索引
      ALTER TABLE `t_role` ADD UNIQUE(`role_code`);
      • 新增数据
      insert into t_role(role_code,role_name) values('ADMIN','管理员');
insert into t_role(role_code,role_name) values('SUPER_ADMIN','超级管理员');
insert into t_role(role_code,role_name) values('TEST','测试角色');
      • 删除数据
      DELETE FROM t_role WHERE role_id = 3 ;
      • 修改数据
      UPDATE t_role set ROLE_NAME = '超级管理员' WHERE ROLE_ID = 2
      • 查询数据
      select * from t_role ;
select * from t_role where role_id = 2;

    • 资源表

      • 创建表
      DROP TABLE IF EXISTS `t_resource`;
CREATE TABLE `t_resource` (
  `resource_id` BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '资源标识',
  `resource_name` varchar(100) NOT NULL COMMENT '资源名称',
  `resource_type` varchar(10) NOT NULL COMMENT '资源类型',
  `parent_id` BIGINT DEFAULT NULL COMMENT '上级资源ID',
  `status` TINYINT NOT NULL DEFAULT 1 COMMENT '资源状态',
  `url` varchar(255) DEFAULT NULL COMMENT '资源路径',
  `order` INT DEFAULT 0 COMMENT '序号',
  `create_time` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  `modify_time` timestamp NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT '资源表';
      • 新增数据
      	insert into t_resource(resource_name,resource_type ,`order`,url) values ('用户管理','MENU',1,'/user');
insert into t_resource(resource_name, resource_type ,parent_id,`order`,url) values ('DETAIL','BUTTON',1,1,'/user/{id}';
insert into t_resource(resource_name, resource_type ,parent_id,`order`,url) values ('ADD','BUTTON',1,2,'/user');
insert into t_resource(resource_name, resource_type ,parent_id,`order`,url) values ('EDIT','BUTTON',1,3,'/user');
insert into t_resource(resource_name, resource_type ,parent_id,`order`,url) values ('DELETE','BUTTON',1,4,'/user');
      • 查询数据
      select * from t_resource where parent_id is null;
  select * from t_resource where parent_id = 1 order by `order`;
select * from t_resource ;

    • 用户角色关联表

      • 创建表
      DROP TABLE IF EXISTS `t_user_role_rel`;
CREATE TABLE `t_user_role_rel` (
`id` BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT 'ID',
`user_id` BIGINT NOT NULL,
`role_id` BIGINT NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT '用户角色关联表';
      • 新增数据
      insert into t_user_role_rel(user_id, role_id) values (1,2);
      • 修改数据
      update t_user_role_rel set role_id = 1 where id = 1;
      • 删除数据
      delete from t_user_role_rel where id = 1;
      • 查询数据
      select * from t_user_role_rel;

select tu.user_id,tr.role_id,tu.user_name,tr.role_code,tr.role_name from t_user_role_rel
left join t_user tu on t_user_role_rel.user_id = tu.user_id
left join t_role tr on t_user_role_rel.role_id = tr.role_id

    • 角色资源关联表

      • 创建表
      DROP TABLE IF EXISTS `tb_admin_role_menu_rel`;
 CREATE TABLE `tb_admin_role_menu_rel` (
  `role_id` int(11) NOT NULL,
  `menu_id` varchar(50) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
      • 创建索引
      # 新增联合唯一索引
ALTER TABLE t_role_resource_rel ADD UNIQUE ROLEID_RESOURCEID_UNIQUE(role_id,resource_id);
# 新增联合唯一索引另一种方式
CREATE UNIQUE INDEX ROLEID_RESOURCEID_UNIQUE ON t_role_resource_rel(role_id,resource_id);
# 查看表当前索引
show index from t_role_resource_rel;
# 删除索引
ALTER TABLE t_role_resource_rel drop INDEX ROLEID_RESOURCEID_UNIQUE;
      • 新增数据
      insert into t_role_resource_rel(role_id, resource_id) values (1,1);
insert into t_role_resource_rel(role_id, resource_id) values (1,2);
insert into t_role_resource_rel(role_id, resource_id) values (1,3);
insert into t_role_resource_rel(role_id, resource_id) values (1,4);
      • 删除数据
      DELETE FROM t_role_resource_rel where role_id = 1 AND resource_id = 2;
      • 查询数据
      # 查询角色权限
select tr.role_code,tr.role_name,t.resource_name,t.resource_type,t.parent_id,t.url from t_role_resource_rel left join t_role tr on t_role_resource_rel.role_id = tr.role_id left join t_resource t on t_role_resource_rel.resource_id = t.resource_id where tr.role_id = 1;

# 查询用户有哪些权限
select tu.user_name,tr.role_code,tr.role_name,r.resource_name,r.resource_type,r.parent_id,r.url from t_resource r
    left join t_role_resource_rel trrr on r.resource_id = trrr.resource_id
    left join t_role tr on trrr.role_id = tr.role_id
    left join t_user_role_rel turr on tr.role_id = turr.role_id
    left join t_user tu on turr.user_id = tu.user_id
where tu.user_id = 1 and tu.status = 1
